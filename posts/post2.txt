<script>
    // Function to load content from a specific text file
    function loadTextFile(filename) {
        var xhr = new XMLHttpRequest();
        xhr.open('GET', filename, true);

        xhr.onreadystatechange = function () {
            if (xhr.readyState === 4 && xhr.status === 200) {
                document.getElementById('text-container').innerText = xhr.responseText;
            }
        };

        xhr.send();
    }

    // Function to get all text files in the current directory
    function loadAllTextFiles() {
        var xhr = new XMLHttpRequest();
        xhr.open('GET', '.', true);

        xhr.onreadystatechange = function () {
            if (xhr.readyState === 4 && xhr.status === 200) {
                var fileNames = xhr.responseText.split('\n');
                // Display the content of each text file in order
                var currentIndex = 0;

                function cycleTextFiles() {
                    loadTextFile(fileNames[currentIndex]);
                    currentIndex = (currentIndex + 1) % fileNames.length;
                }

                // Initial load
                cycleTextFiles();

                // Set interval to update content every 5 seconds (adjust as needed)
                setInterval(cycleTextFiles, 5000);
            }
        };

        xhr.send();
    }

    // Load all text files in the current directory
    loadAllTextFiles();
</script>
